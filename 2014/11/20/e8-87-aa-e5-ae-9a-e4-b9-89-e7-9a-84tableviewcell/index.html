#import <UIKit/UIKit.h>

#import “ASJsonFriendListModel.h”

#import “ASJsonNearbyModel.h”

typedef NS_ENUM(NSInteger, ASSendPhotoType) {

ASSendPhotoNearby = 0,
ASSendPhotoFriend,
ASSendPhotoNearbyAll,
ASSendPhotoFriendAll

};

@interface ASSendPhotoCell : UITableViewCell

{
weak IBOutlet UIImageView *avatar; weak IBOutlet UILabel name;
__weak IBOutlet UILabel about;
__weak IBOutlet UIButton *checkBox;

__weak IBOutlet UIView *friendView;
__weak IBOutlet UIImageView *gender;
__weak IBOutlet UILabel *distance;

}

@property (copy,nonatomic) NSString *imageName;

  • (ASSendPhotoCell *)cellfromXibWithType:(ASSendPhotoType)type;

    //  ASSendPhotoCell.m

    //  Copyright (c) 2014年 asan. All rights reserved.
    //

#import “ASSendPhotoCell.h”

@implementation ASSendPhotoCell

  • (void)awakeFromNib {
    // Initialization code
    }

  • (void)setSelected:(BOOL)selected animated:(BOOL)animated {
    [super setSelected:selected animated:animated];
    // Configure the view for the selected state
    }

  • (void)setImageName:(NSString *)imageName
    {
    [checkBox setImage:[UIImage imageNamed:imageName] forState:UIControlStateNormal];
    }

  • (ASSendPhotoCell )cellfromXibWithType:(ASSendPhotoType)type
    {
    NSArray     nibs = [[NSBundle mainBundle] loadNibNamed:@”ASSendPhotoCell” owner:nil options:nil];
    for (id v in nibs) {
    if ([v isKindOfClass:[ASSendPhotoCell class]]) {
    ASSendPhotoCell *cell = (ASSendPhotoCell *)v;
    [cell setUpSubviews:type];
    [cell setUpFonts];
    return cell;
}
    }
    return nil;

}

  • (void)setUpSubviews:(ASSendPhotoType)type
    {
    if (type == ASSendPhotoNearby) {
    about.hidden = YES;
friendView.hidden = NO;
    }else {
    friendView.hidden = YES;
about.hidden = NO;
    }
    avatar.layer.cornerRadius = avatar.height * 0.5;
    avatar.clipsToBounds = YES;
    }
  • (void)setUpFonts
    {

}

  • (void)loadCellWithData:(id)data byType:(ASSendPhotoType)type
    {
    if (type == ASSendPhotoNearby) {
    ASJsonNearbyModelData *_data = (ASJsonNearbyModelData *)data;
[self loadAvatar:_data.avatar];
[self loadName:_data.name];
[self loadNearby:_data.gender age:_data.age distance:_data.distance];
    }else if (type == ASSendPhotoFriend){
    ASJsonFriendListData *_data = (ASJsonFriendListData *)data;
[self loadAvatar:_data.avatar];
[self loadName:_data.name];
[self loadAbout:_data.about];
    }else if (type == ASSendPhotoNearbyAll){
    avatar.image = [UIImage imageNamed:@"AvatarPlaceholder"];
[self loadAbout:@"..."];
[self loadName:@"附近好友"];
    }else if (type == ASSendPhotoFriendAll){
    avatar.image = [UIImage imageNamed:@"AvatarPlaceholder"];
[self loadAbout:@"..."];
[self loadName:@"全部好友"];
    }

}

  • (void)loadAvatar:(NSString *)url
    {
    if (!url) {
    return;
    }
    [avatar sd_setImageWithURL:[NSURL URLWithString:[PATH_HOST stringByAppendingString:url]]];
    }
  • (void)loadName:(NSString *)string
    {
    if (!string) {
    return;
    }
    name.text = string;
    }

  • (void)loadAbout:(NSString *)string
    {
    if (!string) {

    return;

    }
    about.text = [string isEqualToString:@””]?@”这个人很懒，介绍都没有”:string;
    }

  • (void)loadNearby:(NSString )_gender age:(NSString )_age distance:(NSString *)_distance
    {
    if (!_gender ||!_age ||_distance) {

    return;

    }
    //性别
    NSString *sex = nil;
    if ([_gender integerValue] == 1) {

    sex = @"man-small";

    }else if([_gender integerValue] == 0){

    sex = @"woman-small";

    }
    gender.image = [UIImage imageNamed:sex];

    NSString ages = [NSString stringWithFormat:@”%@岁 | “,_age];
    NSString     distances = [NSString stringWithFormat:@”%.1fkm”,[_distance floatValue]/1000];
    distance.text = [NSString stringWithFormat:@”%@%@”,ages,distances];
    }
    @end

    - (UITableViewCell )tableView:(UITableView )tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath
    {
      //…示例
      if (indexPath.section == 0) {

    ASSendPhotoCell *cell = [ASSendPhotoCell cellfromXibWithType:ASSendPhotoFriendAll];
[cell loadCellWithData:nil byType:ASSendPhotoFriendAll];
[self cell:cell selected:_bAllFriends];
return cell;

    }else{

    ASSendPhotoCell *cell = [ASSendPhotoCell cellfromXibWithType:ASSendPhotoFriend];
ASJsonFriendListData *data = _friendUsers[indexPath.row] ;
[cell loadCellWithData:data byType:ASSendPhotoFriend];
[self cell:cell selected:[_sFriendsList containsObject:data.paixinID]];
return cell;

    };
    return nil;
    }
     

以上是一个简单的自定义Cell，关于返回Cell的这个静态方法，最好还是要抽出来，写一个category，不然每次新建Cell都需要添加这个方法，略微累赘，我要向优雅的代码看齐~ 添加了autolayout约束条件，控制台打印出了一些提示信息。之前都是去除勾选的，所以最近才开始学习怎么使用autolayout，对于6plus的适配必须得上这个利器了，学会了这玩意，以后再多的屏幕尺寸也不虚了，想想就有点小鸡冻呢。
